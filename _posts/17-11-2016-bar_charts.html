    x = ['Nuclear', 'Hydro', 'Gas', 'Oil', 'Coal', 'Biofuel']
    energy = [5, 6, 15, 22, 24, 8]

    import c3pyo as c3
    chart = c3.BarChart()
    chart.plot(energy, color='g', label='Energy from fuel source')
    chart.set_xticklabels(x)
    chart.ylabel('Energy Output (GJ)')
    chart.show()
